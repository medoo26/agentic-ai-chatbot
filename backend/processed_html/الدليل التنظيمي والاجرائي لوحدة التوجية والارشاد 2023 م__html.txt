<h2>الفصل الأول</h2>
<h3>المادة الأولى: التعريفات</h3>
<p>الجامعة: هي جامعة الأمير سطام بن عبدالعزيز.</p>
<p>وحــدة التوجيــه والإرشــاد: هــي وحــدة التوجيــه والإرشــاد فــي عمــادة شــؤون الطــلاب بجامعــة الأميــر ســطام بــن عبدالعزيــز، وتعنــى بتقديــم خدمــات الإرشــاد والمســاعدة المتخصصــة (الاجتماعيــة، والنفســية، والأكاديميــة، والمهنــية) للطــلاب والطالبــات فــي بيئــة آمنــة وداعمــة.</p>
<p>وحــدة الإرشــاد الجامعــي: هــي وحــدة الإرشــاد الجامعــي فــي الكليــة، وتعنــى بتوجيــه الطــلاب والطالبــات توجيــاً أكاديميــاً وعليمــاً، وتهيئتهــم نفســياً وتربويــاً. وتتولــى الوحــدة تحويــل الطــلاب الذيــن لديهــم مشــكلات ذات جــذور نفســية واجتماعيــة إلــى وحــدة التوجيــه والإرشــاد فــي عمــادة شــؤون الطــلاب.</p>
<p>الطــلاب: هــم طــلاب جامعــة الأميــر ســطام بــن عبدالعزيــز المقيــدون فــي مرحلــة البكالوريــوس- الماجســتير- الدكتــوراه.</p>
<p>مشــرف وحــدة الإرشــاد الجامعــي: هــو عضــو فاعــل، ويشــرف علــى الوحــدة ويعمــل علــى تقديــم الخدمــات والإحــالة للطــلاب.</p>
<p>المرشد: هو الشخص المختص الذي يقوم بتقديم خدمات الإرشاد والتوجيه.</p>
<p>المسترشــد: هــو الطالــب أو الطالبــة الــذي لديــه مشــكلات اجتماعيــة أو نفســية أو أســرية أو مهنيــة أو ســوء توافــق مــع الآخريــن، ويرغــب فــي الحصــول علــى مســاعدة للتغلــب علــى تلــك المشــكلات.</p>
<p>الإرشاد الفردي: العملية الإرشادية التي تخدم مسترشداً واحداً لديه مشكلات أو يمر بظروف خاصة به.</p>
<p>الإرشــاد الجمعــي: هــي العمليــة الإرشــادية التــي تخــدم عــددا مــن المسترشــدين بيــن علــى أن تكــون لديهــم نفــس المشــكلة.</p>
<p>الإرشــاد النفســي: هــو تقديــم خدمــات التكيــف النفســي والتوجيــه المناســب حــول كيفيــة التعامــل مــع المشــكلات النفســية التــي يمكــن أن تؤثــر بشــكل ســلبي علــى مشــوار الطــلاب الدراســي، وتطويــر بعــض مهــارات حــل المشــكلات التــي تســاعد الطــلاب علــى التعامــل مــع قضايــا معينــة تحيــط بحياتهــم مســتقبلاً.</p>
<p>الإرشــاد الاجتماعــي: هــو عمليــة مشــتركة تهتــم بمســاعدة مسترشــد، أو عــدد مــن المسترشــدين علــى التكيــف الاجتماعــي وتنميــة قدرتــه علــى بنــاء علاقــات إيجابيــة مــع زملائــه والمحيطيــن بــه وأعضــاء هيئــة التدريــس بصــورة فاعلــة، وكذلــك تعزيــز التواصــل بيــن الحيــاة الجامعيــة وامتداداتهــا فــي المجتمــع.</p>
<p>الإرشــاد المهنــي: هــو عمليــة مشــتركة تتــم بيــن المرشــد والمسترشــد لمســاعدته فــي التعــرف علــى ميولــه واســتعداداته وقدراتــه ممــا يمكنــه مــن اختيــار تخصــص مناســب، ومهنــة مســتقبلية مناســبة.</p>
<p>اســتمارة الاستشــارة (دراســة الحالــة): هــو نمــوذج يســتخدم لجمــع كافــة المعلومــات حــول المسترشــد (نمــوذج رقم 1).</p>
<p>ملــف المسترشــد: هــو الملــف الخــاص بالطالــب فــي الوحــدة والــذي يتضمــن جميــع المعلومــات المتعلقــة بالحالــة.</p>
<p>اســتمارة طلــب الحصــول علــى الخدمــة (نمــوذج الطالــب): هــو نمــوذج يتــم تعبئتــه مــن قبــل الطالــب للحصــول علــى الخدمــة مــن قبــل وحــدة الإرشــاد الجامعــي فــي الكليــة (نمــوذج رقم 2).</p>
<p>اســتمارة الإحالــة (نمــوذج عضــو هيئــة التدريــس): هــو نمــوذج يتــم تعبئتــه مــن قبــل عضــو هيئــة التدريــس لتحويــل الطالــب الــذي يحتــاج إلــى الخدمــة إلــى وحــدة الإرشــاد الجامعــي فــي الكليــة (نمــوذج رقم 3).</p>
<p>البرامــج التوجيهيــة: هــي البرامــج المقدمــة كالإرشــاد الإنمائــي والإرشــاد الوقائــي، وغيرهــا مــن البرامــج والأنشــطة.</p>
<h2>الفصل الثاني</h2>
<h3>المادة الثانية: السياسة العامة لخدمات التوجيه والإرشاد</h3>
<p>تــؤدي خدمــات التوجيــه والإرشــاد دوراً مهمــاً فــي تعزيــز صحــة الطــلاب النفســية وخلــق بيئــة إيجابيــة تدعــم انســجامهم مــع الحيــاة الجامعيــة وكافــة متطلباتهــا. وتأتــي أهميــة خدمــات التوجيــه والإرشــاد فــي ارتباطهــا بتمكيــن الطــلاب مــن التعلــم واكتســاب المهــارات التــي يحتاجونهــا والتغلــب علــى التحديــات التــي تواجههــم خــلال المرحلــة الجامعيــة. إن عمليــة التوجيــه والإرشــاد عمليــة متممــة ومكملــة لعمليــة التعليــم والتعلــم، حيــث إنــهــا تعطــي للعمليــة التربويــة دافعــاً لتجعلهــا أكثــر فاعليــة.</p>
<p>أولاً: سياسة عدم التمييز:</p>
<p>تمتثــل جامعــة الأميــر ســطام بــن عبدالعزيــز بمبــدأ عــدم التمييــز الــذي يعطــي الحــق لجميــع الطــلاب فــي الحصــول علــى جميــع الخدمــات التــي تقدمهــا وحــدة التوجيــه والإرشــاد ووحــدات الإرشــاد الجامعــي فــي الكليــات.</p>
<p>ثانياً: تقديم الخدمات:</p>
<p>يحصــل الطــلاب المقيــدون فــي الجامعــة علــى خدمــات التوجيــه والإرشــاد الاجتماعــي، والنفســي، والمهنــي، والأكاديمــي.</p>
<p>ثالثاً: سرية المعلومات:</p>
<p>تتعهد وحدة التوجيه والإرشاد بمن يمثلها من رؤساء وأخصائيين وإداريين بالمحافظة على سرية المعلومات التــي تتعلــق بالطــلاب، وحفــظ جميــع التقاريــر والمســتندات، علــى أن يقتصــر اســتعمال هــذه المعلومــات مــن قبــل الوحــدة وعندمــا يتــم تحويــل الطالــب إلــى جهــة خارجيــة فيرفــق ملــف الإحالــة مــع إشــعار ذي طابــع ســري.</p>
<p>ســنوات بعــد التأكــد مــن أن الطالــب غيــر مســجل بالجامعــة، مــن خــلال لجنــة وبمحضــر رســمي.</p>
<p>رابعاً: أخلاقيات المرشد:</p>
<p>الأخــلاق والمبــادئ العامــة: أن يتحلــى المرشــد بالأخــلاق الإســلامية قــولاً وعمــلاً وأن يكــون قــدوة حســنة فــي الصبــر والأمانــة وتحمــل المســؤولية مــع مراعــاة التعامــل بالمرونــة وعــدم التقيــد بأســاليب محــددة فــي التعامــل مــع المسترشــدين.</p>
<p>الســرية: أنــه مــن واجــب وأمانــة المرشــد الحفــاظ علــى ســرية المعلومــات، وألا يبــاح بــأي معلومــة إلا بموافقــة المسترشــد، ومــا تقتضيــه المصلحــة مــن الحفــاظ علــى الأرواح والممتلــكات.</p>
<p>العلــم والمعرفــة: أن يكــون المرشــد مؤهــلاً بالعلــم والمعرفــة والخبــرات اللازمــة للإرشــاد وأن يكــون علــى وعــي بذاتــه وحاجاتــه ودوافعــه وعــدم إســقاطها فــي أثنــاء العمليــة الإرشــادية.</p>
<p>العلاقــة المهنيــة: أن تكــون العلاقــة الإرشــادية بيــن المرشــد والمسترشــد علاقــة مهنيــة فــي إطــار المعاييــر الأخلاقيــة والاجتماعيــة، وألا تتطــور إلــى أي نــوع مــن العلاقــات. وألا يســتخدم المرشــد أي نــوع مــن أجهــزة التســجيل إلا بعــد موافقــة المسترشــد.</p>
<h2>الفصل الثالث</h2>
<h3>المادة الثالثة: وحدة الإرشاد الجامعي في الكلية</h3>
<p>بتوجيــه الطــلاب والطالبــات توجيهــاً أكاديميــاً وعليمــاً، وتهيئتهــم نفســياً وتربويــاً. وتتولــى الوحــدة تحويــل الطــلاب الذيــن لديهــم مشــكلات ذات جــذور نفســية واجتماعيــة إلــى وحــدة التوجيــه والإرشــاد فــي عمــادة شــؤون الطــلاب وفــق نمــوذج الإحالــة (نمــوذج رقم 4).</p>
<p>مهام الوحدة:</p>
<p>الكشف عن الحالات التي تحتاج إلى خدمات التوجيه والإرشاد في الكلية.</p>
<p>تهيئة الطلاب والطالبات للاندماج والتوافق التام مع البيئة الجامعية.</p>
<p>تعريــف الطــلاب والطالبــات بأنظمــة وقوانيــن الجامعــة، وبالمســتجد مــن القــرارات والتعديــلات المســتحدثة، وحــل مشــكلات القبــول والمشــكلات الدراســية وخاصــةً للطلبــة الجــدد بحيــث تســهل عمــل الطالــب الجامعــي.</p>
<p>تقديم البرامج والدورات وورش العمل للطلاب.</p>
<p>دراسة الظواهر والمشكلات السلوكية التي تظهر في الكلية.</p>
<p>مساعدة الطلاب على التعرف على ميولهم واستعداداتهم وقدراتهم وسماتهم الشخصية.</p>
<p>رفع التقارير الدورية لوحدة التوجيه والإرشاد في عمادة شؤون الطلاب.</p>
<p>إحالــة الطلبــة الذيــن لديهــم مشــكلات ذات جــذور نفســية واجتماعيــة إلــى وحــدة التوجيــه والإرشــاد فــي عمــادة شــؤون الطــلاب.</p>
<h3>المادة الرابعة: وحدة التوجيه والإرشاد في عمادة شؤون الطلاب</h3>
<p>وحــدة التوجيــه والإرشــاد فــي عمــادة شــؤون الطــلاب هــي المســؤولة عــن تطويــر الأنظمــة والإجــراءات الخاصــة بخدمــات التوجيــه والإرشــاد وتنفيــذ البرامــج التوجيهيــة لطــلاب الجامعــة وطالباتهــا. كذلــك اســتقبال الحــالات المحولــة لهــا.</p>

<h2>المادة الرابعة: وحدة التوجيه والإرشاد في عمادة شؤون الطاب</h2>
<p>وحــدة التوجيــه والإرشــاد فــي عمــادة شــؤون الطــلاب هــي المســؤولة عــن تطويــر الأنظمــة والإجــراءات الخاصــة بخدمــات التوجيــه والإرشــاد وتنفيــذ البرامــج التوجيهيــة لطــلاب الجامعــة وطالباتهــا. كذلــك اســتقبال الحــالات المحولــة لهــا مــن قبــل وحــدات الإرشــاد الجامعــي فــي الكليــات لتقديــم الخدمــات اللازمــة لهــم. كذلــك تقــدم الوحــدة مجموعــة مــن الــدورات وورش العمــل لمشــرفي وحــدات الإرشــاد الجامعــي داخــل الكليــات.</p>
<h3>المادة 1</h3>
<p>مهام الوحدة:</p>
<p>التواصــل مــع وحــدات الإرشــاد الجامعــي داخــل الكليــات للتعــرف علــى كل الحــالات وإجــراء التســهيلات اللازمــة لحــل مشــكلاتهم وتقديــم الدعــم الــلازم لهــم.</p>
<p>تحديد الاحتياجات الخاصة بخدمات التوجيه والإرشاد في الجامعة.</p>
<p>توفير الأجهزة والأدوات المساعدة لتقديم خدمات التوجيه والإرشاد.</p>
<p>إنشاء قاعدة بيانات خاصة للطلاب المستفيدين من خدمات التوجيه والإرشاد.</p>
<p>تهيئة الطلاب والطالبات للاندماج والتوافق التام مع البيئة الجامعية.</p>
<p>تقديم البرامج والدورات وورش العمل لمشرفي وحدات الإرشاد الجامعي.</p>
<p>دراسة الظواهر والمشكلات السلوكية التي تظهر في الجامعة.</p>
<p>إعداد وتحديث الأدلة التنظيمية والإجرائية لخدمات التوجيه والإرشاد في الجامعة.</p>
<p>تنفيذ البرامج التوجيهية ونشر الوعي بين الطلاب وأبناء المجتمع بطرق تعزيز الصحة النفسية وأهميتها.</p>
<p>عقد الشراكات مع الجهات خارج الجامعة لرفع مستوى خدمات التوجيه والإرشاد.</p>
<h3>المادة 2</h3>
<p>يتم طلب الخدمة من خلال:</p>
<p>يتقــدم الطالــب للحصــول علــى خدمــة التوجيــه والإرشــاد مــن خــلال تعبئــة اســتمارة طلــب الحصــول علــى الخدمــة.</p>
<p>لــدى وحــدة الإرشــاد الجامعــي فــي الكليــة التــي يــدرس بهــا.</p>
<p>تحويــل الطالــب مــن قبــل عضــو هيئــة التدريــس، أو المرشــد الأكاديمــي بالكليــة إلــى وحــدة الإرشــاد الجامعــي.</p>
<p>من خــلال تعبئــة اســتمارة الإحالــة.</p>
<p>بعــد ذلــك يقــوم المشــرف علــى وحــدة الإرشــاد الجامعــي بمراجعــة اســتمارة طلــب الخدمــة المقدمــة مــن الطالــب، أو الإحالــة مــن قبــل عضــو هيئــة التدريــس بســرية تامــة للتعــرف علــى المشــكلة.</p>
<p>وبعــد دراســة الاســتمارة، يتــم الاتصــال بالطالــب لطلــب أي معلومــات إضافيــة وتحديــد موعــد للاجتمــاع معــه.</p>
<p>وخــلال الاجتمــاع الأول يقــوم المشــرف بتنفيــذ دراســة الحالــة تتضمــن جميــع المعلومــات اللازمــة.</p>
<p>وبعــد دراســة المعلومــات بالتفصيــل، يقــوم المشــرف فــي الاجتمــاع الثانــي بشــرح كامــل لتفاصيــل الخطــة المعــدة للطالــب والخدمــات المقدمــة خــلال فتــرة تقديــم الخدمــات.</p>
<p>وفــي حــال وجــود حــالات ذات جــذور نفســية واجتماعيــة التــي لا يكــن تقديــم الخدمــات لهــا مــن خــلال وحــدة الإرشــاد الجامعــي، يتــم إحالــة الطالــب إلــى وحــدة التوجيــه والإرشــاد فــي عمــادة شــؤون الطــلاب لاســتكمال باقــي الإجــراءات الإرشــادية المناســبة وفــق نمــوذج الإحالــة.</p>
<h2>الفصل الرابع</h2>
<h2>الخدمات</h2>
<h3>المادة السادسة: الخدمات</h3>
<p>خدمــات التوجيــه والإرشــاد تقــدم مــن خــلال وحــدة الإرشــاد الجامعــي فــي الكليــات ووحــدة التوجيــه والإرشــاد عمــادة شــؤون الطــلاب وفــق الإجــراءات المنظمــة لخدمــات التوجيــه والإرشــاد والتــي يمكــن تصنيفهــا إلــى فــي أربــع فئــات:</p>
<p>الاستشارات النفسية هي لقــاء يتــم بيــن المرشــد والمسترشــد الهــدف منه الوصــول إلى حل لمشــكلة يعاني منهــا المسترشــد كاضطراب نفسي بســيط مثــل القلــق، أو الضغــط النفســي، أو مشــكلة شــخصية لا يســتطيع بمفــرده مواجهاتهــا أو التغلــب عليهــا وحلهــا.</p>
<p>الاستشارات الاجتماعية هي استشــارة تقــدم للمسترشــد لمســاعدته علــى التكيــف مــع المشــكلات الاجتماعيــة التــي قــد يعانــي منهــا.</p>
<p>الاستشارات المهنية هي الاستشــارة التــي تقــدم للمسترشــد الــذي لايــزال متــردداً باختيــار مهنــة المســتقبل.</p>
<p>الاستشارات الأكاديمية هي استشــارة تهــدف إلــى مســاعدة الطالــب علــى اكتشــاف قدراتــه وإمكاناتــه بهــدف معاونتــه فــي اتخــاذ القــرارات التــي تتصــل بخطتــه الدراســية.</p>
